Here’s a **simple yet secure example of cookie-based authentication in NestJS**, including:

* User login with cookies
* Signed, `HttpOnly` cookies
* Middleware/guard to protect routes
* Logout functionality

---

## ✅ **1. Install Dependencies**

```bash
npm install cookie-parser
npm install @types/cookie-parser --save-dev
```

---

## ✅ **2. Enable `cookie-parser` in `main.ts`**

```ts
// main.ts
import * as cookieParser from 'cookie-parser';
import { NestFactory } from '@nestjs/core';
import { AppModule } from './app.module';

async function bootstrap() {
  const app = await NestFactory.create(AppModule);
  app.use(cookieParser('your-secret-key')); // used for signed cookies
  await app.listen(3000);
}
bootstrap();
```

---

## ✅ **3. Auth Controller Example**

```ts
// auth.controller.ts
import { Controller, Post, Req, Res, Get } from '@nestjs/common';
import { Response, Request } from 'express';

@Controller('auth')
export class AuthController {
  @Post('login')
  login(@Res({ passthrough: true }) res: Response) {
    // Replace this with real user verification
    const userId = 'user123';

    res.cookie('auth_token', userId, {
      httpOnly: true,
      signed: true,
      sameSite: 'lax',
      secure: process.env.NODE_ENV === 'production',
      maxAge: 1000 * 60 * 60, // 1 hour
    });

    return { message: 'Logged in' };
  }

  @Post('logout')
  logout(@Res({ passthrough: true }) res: Response) {
    res.clearCookie('auth_token');
    return { message: 'Logged out' };
  }

  @Get('me')
  getProfile(@Req() req: Request) {
    const userId = req.signedCookies['auth_token'];
    if (!userId) {
      return { message: 'Not authenticated' };
    }

    // Simulate DB lookup
    return { user: { id: userId, name: 'John Doe' } };
  }
}
```

## ✅ **4. Auth Guard (Optional)**

To protect routes globally or per-controller:

```ts
// auth.guard.ts
import { CanActivate, ExecutionContext, Injectable } from '@nestjs/common';
import { Request } from 'express';

@Injectable()
export class AuthGuard implements CanActivate {
  canActivate(context: ExecutionContext): boolean {
    const req = context.switchToHttp().getRequest<Request>();
    const userId = req.signedCookies['auth_token'];
    return !!userId;
  }
}
```

Use like this:

```ts
@UseGuards(AuthGuard)
@Get('protected')
getProtectedData() {
  return { secret: 'You are authenticated!' };
}
```

---

## 🛡️ Security Tips

* Always use `HttpOnly` and `Secure` in production
* Use `SameSite='Lax'` or `'Strict'` to mitigate CSRF
* Store only non-sensitive IDs in cookies
* Use CSRF tokens if you allow state-changing requests from browsers
